```javascript
function cake(x, y) {
  y = y.split('');
  let sum1 = 0;
  let sum2 = 0;
  var alphabet = 'abcdefghijklmnopqrstuvwxyz'.split('');
  for (let i = 0; i < y.length; i++) {
    if (i % 2 == 0) {
      let x = y[i].charCodeAt();
      sum1 += x;
    }
    if (i % 2 != 0) {
      let val = y[i].toLowerCase();
      let z = alphabet.indexOf(val) + 1;
      sum2 += z;
    }
  }
  let sum = sum1 + sum2;
  if (sum > x * 0.7) {
    return 'Fire!';
  }
  return 'That was close!';
}
```
